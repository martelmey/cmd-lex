Get-NetIPInterface {
    * https://learn.microsoft.com/en-us/powershell/module/nettcpip/get-netipinterface?view=windowsserver2022-ps

    -AutomaticMetric # assigns lowest metric to fastest interface automatically
    
    examples {
        # get ip info for machine
        Get-NetIPInterface | Format-Table
        Get-NetIPInterface | Sort-Object -Property InterfaceIndex | Format-Table
        # get ip info by interface index
        Get-NetIPInterface -InterfaceIndex 12
    }
}
