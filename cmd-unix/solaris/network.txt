nc {
    -4                         Use IPv4 only
    -c, --sh-exec <command>    Executes the given command via /bin/sh
    -e, --exec <command>       Executes the given command
    -d, --delay <time>         Wait between read/writes
    -o, --output <filename>    Dump session data to a file
    -p, --source-port port     Specify source port to use
    -s, --source addr          Specify source address to use (doesn't affect -l)
    -l, --listen               Bind and listen for incoming connections
    -k, --keep-open            Accept multiple connections in listen mode
    -n, --nodns                Do not resolve hostnames via DNS
    -t, --telnet               Answer Telnet negotiations
    -u, --udp                  Use UDP instead of default TCP
    -v, --verbose              Set verbosity level (can be used several times)
    -w, --wait <time>          Connect timeout
    -z                         Zero-I/O mode, report connection status only
    --append-output        Append rather than clobber specified output files
    --send-only            Only send data, ignoring received; quit on EOF
    --recv-only            Only receive data, never send anything
    --proxy <addr[:port]>  Specify address of host to proxy through
    --proxy-type <type>    Specify proxy type ("http" or "socks4" or "socks5")
    --proxy-auth <auth>    Authenticate with HTTP or SOCKS proxy server
    
    examples {
        # test port
        netcat -z -n -v 192.168.60.10 22
    }
}

tcpdump {
    https://docs.oracle.com/cd/E88353_01/html/E72487/tcpdump-8.html

    tcpdump [ -AbdDefhHIJKlLnNOpqStuUvxX# ] [ -B buffer_size ]
    [ -c count ]
    [ -C file_size ] [ -G rotate_seconds ] [ -F file ]
    [ -i interface ] [ -j tstamp_type ] [ -m module ] [ -M secret ]
    [ --number ] [ -Q in|out|inout ]
    [ -r file ] [ -V file ] [ -s snaplen ] [ -T type ] [ -w file ]
    [ -W filecount ]
    [ -E spi@ipaddr algo:secret,...  ]
    [ -y datalinktype ] [ -z postrotate-command ] [ -Z user ]
    [ --time-stamp-precision=tstamp_precision ]
    [ --immediate-mode ] [ --version ]
    [ expression ]

    examples {
        # To print all packets arriving at or departing from sundown:
        tcpdump host sundown
        # To print all IP packets between ace and any host except helios:
        tcpdump ip host ace and not helios
    }
}

traceroute {
    -a # all
    -i <interface>
    -r # bypass normal routing tables
    -S # show no. probes unanswered for each hop
    -s <src_ip> # specify source ip on host

    examples {
        -i eno1
        -s 192.168.61.40 142.34.146.251
    }
}
