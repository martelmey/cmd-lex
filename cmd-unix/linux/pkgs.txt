rpm {
    rpm -q, --query
        --configfiles
        --i, info
        --filesbypkg
        -l, --list
        -R, --requires
        -f, --file
        -p, --package
        -v, --verify
        -i, --install
        -U, upgrade
}

yum {
    clean
        packages
        metadata
        headers
        all
	install
		--downloadonly
		--installroot
		--downloaddir
        ./*rpm
        --downloadonly --installroot=/root/libvirt-installroot --downloaddir=/root/libvirt libvirt
	list
		installed
	info
	groups
		group list
		group info
		group summary
	search
		--showduplicates
	resolvedep
	deplist
		--showduplicates
		--verbose
	repolist
		-v
	repoinfo
	history
	--enablerepo
        --enablerepo=offline-libvirt install libvirt
        --enablerepo=offline-virt-viewer install virt-viewer
}

yumdownloader {
	--destdir
	--resolve
	--source
	--archlist=x86_64
    
    --destdir=/root/yumdownloader-libvirt --resolve libvirt
    --destdir=/root/yumdownloader-virt-viewer --resolve virt-viewer
}
