smb {
    # https://www.linuxtechi.com/install-configure-samba-centos-8/
    # svc
    systemctl status {smb,nmb}
        # apply change
        systemctl restart {smb,nmb}
    # fw
    firewall-cmd --add-service=samba
    # cfg
    /etc/samba/smb.conf
        # allow anon
        [global]\n map to guest = bad user
        # new share
        [sharename]
        path = /export/a
        guest only = yes # anon, w/ writeable=no more secure
        # secure only
        [sharename]
        guest ok = no
    # add usr to db
    smbpasswd -a sambauser
}

fdisk {
    Manipulate disk partition table

    fdisk [-uc] [-b sectorsize] [-C cyls] [-H heads] [-S sects] device
    fdisk -l [-u] [device...]
    fdisk -s partition...

    Options:
        -b, --sector-size sectorsize
            Specify sector size of disk.
            512, 1024, 2048, 4096
        -B, --protect-boot
            Don't erase the beginning of first disk sector.
            For GPT and MBR.
        -c, --compatibility[=mode]
            dos, default=nondos
        -l, --list
            list part tables for specified devices.
            default=/proc/partitions
}

find {
    <path to files><*>
    -mtime (days old of file)(+5=files older than 5 days)
    -exec <command>
    {} \; (ends command)

        find /path/to/dir/ -printf '%s %p\n'| sort -nr | head -10
        find . -printf '%s %p\n'| sort -nr | head -10
        find / -name vssat -print

        # print list of directories in directory (no leading slash)
        find /opt/splunk/etc/deployment-apps/ -mindepth 1 -maxdepth 1 -type d -printf '%f\n'
    }
}
