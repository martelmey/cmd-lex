nmap -sn <target addresses>
Results = all responding active hosts
-sn disables port scanning {
	nmap -sn 10.0.0.0/24
}

nmap {
	-V (services)
	-O (OS detection)
	-S (port scan)
}

examples {
	nmap -sS 10.0.0.0/24 (hosts up)
	nmap -sV 10.0.0.0/24 (services)
	nmap -O 10.0.0.0/24 (OS detection)
}
